ABOUT GIT
---------------------------------------------------------------------------------------

git commit -a --amend

# update the forked repo
git remote add upstream git://github.com/whoever/whatever.git # Add the remote, call it "upstream"
# Fetch all the branches of that remote into remote-tracking branches,
# such as upstream/master:
git fetch upstream
git checkout master # Make sure that you're on your master branch
# Rewrite your master branch so that any commits of yours that
# aren't already in upstream/master are replayed on top of that
# other branch:
git rebase upstream/master
git merge upstream/master

git remote show origin   # show a repo's remote url
git remote set-url origin git://new.url.here  # set repo remote url

git pull = git fetch + git merge

ABOUT CI
---------------------------------------------------------------------------------------

Controller:

$this->load->view("news/index", $data);
$this->load->helper("url");  # load helper, then can have the url functions
$this->load->library("form_validation");  # load class, then can $this->form_validation->run() etc.
$this->load->driver();
Model:
parent::__construct();
$this->load->database();
$this->db->get(); $this->db->insert();
$this->input->post("title");  # input class auto called
